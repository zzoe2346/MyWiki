class Solution {
    private static final int MOD = 1234567;
    public int solution(int n) {
        int[] dp = new int[100001];
        dp[0] = 0;
        dp[1] = 1;
        for(int i=2 ; i<=n ; i++){
            dp[i] = (dp[i-1] + dp[i-2])%MOD;
        }
        return dp[n];
    }
}
